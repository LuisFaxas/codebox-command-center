Create `sessions.js` as a new ES module at the project root. Follow the config.js persistence pattern and the research-recommended session shape.

**Session data shape** (per D-01, D-03, D-04):
```javascript
{
  sessionId: string,           // From Claude Code hook
  project: string,             // Project display name
  machine: string,             // Machine identifier (codebox, lenovo, mac)
  cwd: string,                 // Working directory
  status: 'working' | 'done' | 'attention' | 'stale',
  firstSeen: number,           // Date.now() on first upsert
  lastActivity: number,        // Date.now() on every upsert
  lastEventType: string,       // 'done' or 'question'
  eventCount: number,          // Total events for this session
  events: Array<{ type: string, timestamp: number }>,  // Ring buffer
  questionText: string | null, // Per D-03: store even though UI is Phase 6
  questionTimestamp: number | null  // Per D-04: future interaction support
}
```

**Constants:**
- `STALE_MS = 5 * 60 * 1000` (5 minutes — per D-05)
- `REMOVE_MS = 30 * 60 * 1000` (30 minutes — per D-05)
- `PERSIST_INTERVAL_MS = 30000` (30 seconds — research recommendation)
- `CLEANUP_INTERVAL_MS = 30000` (30 seconds — combined with persist)
- `MAX_EVENTS_PER_SESSION = 20` (ring buffer — research recommendation)
- `SESSIONS_FILE = join(DATA_DIR, 'sessions.json')`

**Imports:**
```javascript
import { readFileSync, writeFileSync, renameSync, existsSync } from 'fs';
import { join } from 'path';
import { DATA_DIR } from './config.js';
import { emit } from './sse.js';
```

**Exported functions:**

1. `upsertSession({ sessionId, project, machine, cwd, type, questionText })` — Creates or updates a session. Sets status to `'attention'` if type is `'question'`, otherwise `'done'`. Appends to events ring buffer (shift if > MAX_EVENTS_PER_SESSION). Stores questionText/questionTimestamp when type is 'question'. Calls `emit('session:update', delta)` where delta is the full session object minus the `events` array (per research SSE format recommendation). Returns the session object.

2. `getAllSessions()` — Returns `Object.fromEntries(sessions)` for JSON serialization. Used by GET /sessions endpoint.

3. `getSession(sessionId)` — Returns session or null.

4. `loadSessions()` — Reads `data/sessions.json`, parses JSON, restores only sessions where `Date.now() - session.lastActivity < REMOVE_MS`. Silent catch on any error (missing file, corrupt JSON).

5. `saveSessions()` — Atomic write: `writeFileSync` to `sessions.json.tmp`, then `renameSync` to `sessions.json`. Uses `JSON.stringify({ sessions: Object.fromEntries(sessions), savedAt: Date.now() }, null, 2)`. Silent catch on error.

**Cleanup timer** (combined persist + TTL sweep, runs every 30s):
```javascript
setInterval(() => {
  const now = Date.now();
  for (const [id, session] of sessions) {
    const elapsed = now - session.lastActivity;
    if (elapsed > REMOVE_MS) {
      sessions.delete(id);
      emit('session:remove', { sessionId: id });
    } else if (elapsed > STALE_MS && session.status !== 'stale') {
      session.status = 'stale';
      emit('session:update', sessionDelta(session));
    }
  }
  saveSessions();
}, 30000);
```

**Graceful shutdown** (per research Pitfall 1 — do NOT call process.exit, let other handlers run):
```javascript
function onShutdown() { saveSessions(); }
process.on('SIGTERM', onShutdown);
process.on('SIGINT', onShutdown);
```

**Helper:**
`sessionDelta(session)` — Returns a copy of the session object without the `events` array (to keep SSE payloads small). Include: sessionId, project, machine, cwd, status, firstSeen, lastActivity, lastEventType, eventCount, questionText, questionTimestamp.

```javascript
import { test, expect } from 'playwright/test';

const BASE = 'http://localhost:3099';

test.describe('Notification Regression (NOTIF-01, NOTIF-02, NOTIF-03)', () => {

  test('trigger endpoint accepts POST and returns ok (baseline)', async ({ request }) => {
    const res = await request.post(`${BASE}/trigger`, {
      data: { type: 'done', project: 'RegTest', sessionId: 'reg-001', machine: 'test', cwd: '/tmp' }
    });
    expect(res.ok()).toBeTruthy();
    const body = await res.json();
    expect(body.ok).toBe(true);
  });

  test('NOTIF-03: trigger shows toast in dashboard', async ({ page }) => {
    await page.goto(BASE);
    // Wait for SSE connection
    await page.waitForSelector('.connection-dot.connected', { timeout: 5000 });

    // Fire trigger via API
    await page.request.post(`${BASE}/trigger`, {
      data: { type: 'done', project: 'ToastTest', sessionId: 'reg-002', machine: 'test', cwd: '/tmp' }
    });

    // Toast should appear in #toast-container
    await expect(page.locator('#toast-container .toast')).toBeVisible({ timeout: 5000 });
  });

  test('NOTIF-01: audio endpoint returns WAV after trigger', async ({ request }) => {
    // Trigger to generate cached WAV
    await request.post(`${BASE}/trigger`, {
      data: { type: 'done', project: 'AudioReg', sessionId: 'reg-003', machine: 'test', cwd: '/tmp' }
    });

    // Wait for TTS generation (edge-tts takes 2-4 seconds)
    await new Promise(r => setTimeout(r, 4000));

    const res = await request.get(`${BASE}/notify-wav?type=done&project=AudioReg`);
    expect(res.status()).toBe(200);
    const contentType = res.headers()['content-type'];
    expect(contentType).toMatch(/audio/);
  });

  test('NOTIF-02: VAPID public key endpoint responds (push prerequisite)', async ({ request }) => {
    const res = await request.get(`${BASE}/vapid-public-key`);
    expect(res.ok()).toBeTruthy();
    const body = await res.json();
    expect(body.publicKey).toBeTruthy();
    expect(typeof body.publicKey).toBe('string');
  });

  test('config endpoint returns voice configuration', async ({ request }) => {
    const res = await request.get(`${BASE}/config`);
    expect(res.ok()).toBeTruthy();
    const body = await res.json();
    expect(body.done).toBeTruthy();
    expect(body.done.voice).toBeTruthy();
    expect(body.question).toBeTruthy();
  });
});
```
