**sdk-bridge.js** — Add `findSdkSessionForCwd()` function for session-to-SDK matching (per D-01):

```javascript
export async function findSdkSessionForCwd(targetCwd) {
  try {
    const sessions = await listSessions({ limit: 50 });
    const match = sessions
      .filter(s => s.cwd === targetCwd)
      .sort((a, b) => new Date(b.lastModified) - new Date(a.lastModified));
    return match[0] || null;
  } catch (e) {
    return null;
  }
}
```
